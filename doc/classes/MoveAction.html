                  Class MoveAction

                  Class defined in: src\actions.ts:6

                  Base class for implementation of changing year by clicking on "previous" or "next" arrow. Implements IMoveAction interface and can be overwritten.

                  Example of implementation user's class with move() method. Server returns new data for each year:

                  export class MyMoveAction extends MoveAction {
  move(year: number, data?: Array<IData>): JQueryPromise<any> {
      var deferred = $.Deferred();
      this.getJsonData(year, deferred);
      return deferred.promise();
  }

  private getJsonData(year: number, deferred: JQueryDeferred<any>): void {
      var url = "data/events" + year + ".json";
      $.ajax(url)
          .done((result) => {
              this.data = result;
          })
          .always(() => {
              deferred.resolve();
          });
  }
}
